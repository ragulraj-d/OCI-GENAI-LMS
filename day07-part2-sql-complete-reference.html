-- BASIC SQL COMMANDS
SELECT
Fetch data from a table
SELECT name FROM users;
INSERT INTO
Add new row(s) to a table
INSERT INTO t VALUES(...);
UPDATE
Modify existing rows
UPDATE t SET col=val;
DELETE
Remove rows from a table
DELETE FROM t WHERE id=1;
CREATE TABLE
Define a new table structure
CREATE TABLE t (id INT);
DROP TABLE
Delete table permanently
DROP TABLE users;
ALTER TABLE
Modify table structure
ALTER TABLE t ADD col;
TRUNCATE
Remove all rows, keep structure
TRUNCATE TABLE users;
-- CONSTRAINTS: rules enforced on table columns
PRIMARY KEY
Uniquely identifies each row. NOT NULL + UNIQUE combined
id INT PRIMARY KEY
FOREIGN KEY
Links two tables. Enforces referential integrity
REFERENCES dept(id)
NOT NULL
Column cannot store NULL values
name VARCHAR NOT NULL
UNIQUE
All values in column must be distinct
email VARCHAR UNIQUE
CHECK
Validates a condition before inserting/updating
CHECK (age >= 18)
DEFAULT
Sets a fallback value if none is provided
status DEFAULT 'active'
INDEX
Speeds up search queries on a column
CREATE INDEX ON t(col)

-- CLAUSES: modifiers that shape query results
WHERE
Filters rows based on a condition
WHERE age > 18
HAVING
Filters groups (used with GROUP BY)
HAVING COUNT(*) > 5
FROM
Specifies the source table(s)
FROM employees
JOIN
Combines rows from two tables. Types: INNER, LEFT, RIGHT, FULL
JOIN dept ON t.id=d.id
LIMIT / TOP
Restricts number of rows returned
LIMIT 10
DISTINCT
Returns only unique values
SELECT DISTINCT city
LIKE
Pattern match using % and _ wildcards
WHERE name LIKE 'A%'
IN / BETWEEN
IN: match a list. BETWEEN: range check
WHERE id IN (1,2,3)

-- ALIAS: rename columns or tables temporarily
Column Alias
Renames a column in query output using AS keyword. Does not change actual table.
SELECT first_name AS name,
salary AS monthly_pay
FROM employees;
Table Alias
Shortens table name for readability, especially useful in JOINs and subqueries.
SELECT e.name, d.dept_name
FROM employees AS e
JOIN departments AS d
ON e.dept_id = d.id;
⚡ Key Rules
✓ AS keyword is optional (space works too)
✓ Use quotes for aliases with spaces: AS "Full Name"
✓ Column alias can be used in ORDER BY
✗ Cannot use column alias in WHERE clause
-- ORDER BY: sort the query result set
ASC — Ascending
Sorts from lowest to highest. Default behavior when direction is not specified.
SELECT name, salary
FROM employees
ORDER BY salary ASC;
DESC — Descending
Sorts from highest to lowest. Commonly used for rankings, top N queries.
SELECT name, salary
FROM employees
ORDER BY salary DESC;
Multi-column sort
Sort by department first, then by salary within each department
SELECT name, dept, salary FROM employees
ORDER BY dept ASC, salary DESC;
By column number
ORDER BY 2 DESC
NULL handling
NULLS LAST / FIRST
With alias
ORDER BY alias_name
-- GROUP BY: aggregate rows sharing the same value
How GROUP BY works
Collapses rows with the same value in the grouped column into a single row. Always used with aggregate functions (COUNT, SUM, AVG, etc.).
SELECT dept, COUNT(*) AS total_emp,
AVG(salary) AS avg_salary
FROM employees
GROUP BY dept;
GROUP BY + HAVING
Filter groups after aggregation. WHERE filters rows BEFORE grouping; HAVING filters AFTER.
SELECT dept, COUNT(*) AS cnt
FROM employees
GROUP BY dept
HAVING COUNT(*) > 10;
GROUP BY + ORDER BY
Sort the grouped result. ORDER BY always comes after GROUP BY in the query.
SELECT dept, SUM(salary)
FROM employees
GROUP BY dept
ORDER BY SUM(salary) DESC;
Query execution order
FROM → WHERE → GROUP BY → HAVING → SELECT → ORDER BY
